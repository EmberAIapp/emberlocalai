Vue d'architecture

Principe fondamental

Le GPU MLX et l'ANE ne sont pas en compétition — ils ont des profils d'efficacité complémentaires selon l'arithmetic intensity des opérations. On exploite ça par couche, pas par modèle.

La clé du gain : pipeline overlap. Pendant que l'ANE traite l'attention du bloc N via CoreML, le GPU prépare la projection query/key du bloc N+1 en Metal. La latence effective = max(ANE, GPU) par window, pas sum.

Le gain est conditionnel au coût du transfert IOSurface. Sur M3 Ultra avec GPU à pleine bande (~800 GB/s), la copie vers ANE peut annuler le bénéfice. Sur M2/M3 base, c'est différent — benchmarker avant d'optimiser.

QKV projection, LayerNorm, opérations denses de dimension modérée. 15.8 TOPS sur M3, ~2W. Format NHWC obligatoire, max 16K ops/model.

Le softmax et la full-attention NE tournent PAS efficacement sur ANE. Trop bandwidth-bound. Laisser sur GPU ou utiliser flash attention Metal.

MLXEngine — wrapper GPU

SWIFT Sources/Core/MLXEngine.swift 
import MLX
import MLXNN
import Metal

final class MLXEngine {
    private let device: MTLDevice
    private let cmdQueue: MTLCommandQueue

    // Cache des poids compilés MLX — évite la recompilation JIT
    private var weightCache: [String: MLXArray] = [:]

    init() throws {
        guard let device = MTLCreateSystemDefaultDevice() else {
            throw EngineError.noMetalDevice
        }
        self.device = device
        guard let q = device.makeCommandQueue() else {
            throw EngineError.commandQueueFailed
        }
        self.cmdQueue = q

        // Force MLX à utiliser le même device — unifié
        MLX.setDefaultDevice(.gpu)
    }

    /// FFN forward pass — fused GEGLU + projection
    /// Input: [batch, seq, d_model] float16
    /// Returns: [batch, seq, d_model] float16
    func ffnForward(
        hidden: MLXArray,
        w1: MLXArray,
        w2: MLXArray,
        w3: MLXArray
    ) -> MLXArray {
        // GEGLU : gate × silu(gate) puis projection
        let gate = MLXNN.silu(matmul(hidden, w1.transposed()))
        let up   = matmul(hidden, w3.transposed())
        let act  = gate * up
        return matmul(act, w2.transposed())
    }

    /// Lit un MTLBuffer existant comme MLXArray sans copie
    /// Utilise le backing store de mémoire unifiée
    func wrapBuffer(_ buffer: MTLBuffer, shape: [Int]) -> MLXArray {
        // NOTE: mlx-swift 0.18+ supporte MLXArray(buffer:shape:dtype:)
        // Vérifier l'API exacte selon version — peut nécessiter unsafe ptr
        let ptr = buffer.contents().assumingMemoryBound(to: Float16.self)
        return MLXArray(ptr, shape, dtype: .float16)
    }
}

ANEEngine — wrapper CoreML

SWIFT Sources/Core/ANEEngine.swift 
import CoreML

final class ANEEngine {
    private let model: MLModel
    private let config: MLModelConfiguration

    init(modelURL: URL) throws {
        config = MLModelConfiguration()

        // CRITIQUE : forcer ANE — sinon CoreML peut fallback GPU
        config.computeUnits = .cpuAndNeuralEngine

        // Pré-compiler vers mlmodelc si pas déjà fait
        let compiledURL: URL
        if modelURL.pathExtension == "mlpackage" {
            compiledURL = try MLModel.compileModel(at: modelURL)
        } else {
            compiledURL = modelURL // .mlmodelc déjà compilé
        }

        self.model = try MLModel(contentsOf: compiledURL, configuration: config)
    }

    /// Attention forward (QKV + scaled dot-product)
    /// IMPORTANT: CoreML attend NHWC, shape [1, seq, heads, d_head]
    func attentionForward(
        query: MLMultiArray,
        key:   MLMultiArray,
        value: MLMultiArray
    ) throws -> MLMultiArray {
        let input = try MLDictionaryFeatureProvider(dictionary: [
            "query": query,
            "key":   key,
            "value": value,
        ])
        let output = try model.prediction(from: input)

        guard let result = output.featureValue(for: "attn_output")?.multiArrayValue else {
            throw ANEError.outputMissing
        }
        return result
    }

    /// Version async — utilise MLPredictionOptions pour batch + queue dédiée
    func attentionAsync(
        query: MLMultiArray,
        key: MLMultiArray,
        value: MLMultiArray
    ) async throws -> MLMultiArray {
        let opts = MLPredictionOptions()
        opts.usesCPUOnly = false

        let input = try MLDictionaryFeatureProvider(dictionary: [
            "query": query, "key": key, "value": value
        ])
        return try await Task.detached(priority: .userInitiated) {
            let out = try self.model.prediction(from: input, options: opts)
            return out.featureValue(for: "attn_output")!.multiArrayValue!
        }.value
    }
}

Shared Buffer — Zero-Copy Bridge

Le vrai problème soulevé dans le post original : les transferts IOSurface via kernel calls. La solution : un MTLBuffer avec storageMode .shared comme backing store unique, exposé à CoreML via MLMultiArray(dataPointer:) sans copie.

Cette technique fonctionne sur Apple Silicon (mémoire unifiée). Sur Intel Mac ou si le GPU est discret, la mémoire n'est pas unifiée et la copie est inévitable — à détecter au runtime via MTLDevice.hasUnifiedMemory.
SWIFT Sources/Core/SharedBuffer.swift 
import Metal
import CoreML

/// Buffer partagé Metal ↔ CoreML sans copie intermédiaire.
/// Exploite la mémoire unifiée d'Apple Silicon.
final class SharedBuffer {

    let mtlBuffer: MTLBuffer
    let byteCount: Int
    let shape: [Int]
    let dtype: MLMultiArrayDataType

    // Strides pour MLMultiArray — doit matcher layout Metal (row-major)
    private let strides: [NSNumber]

    init(
        device: MTLDevice,
        shape: [Int],
        dtype: MLMultiArrayDataType = .float16
    ) throws {
        guard device.hasUnifiedMemory else {
            throw SharedBufferError.noUnifiedMemory
        }

        let elementSize = dtype == .float16 ? 2 : 4
        let elementCount = shape.reduce(1, *)
        byteCount = elementCount * elementSize

        // .storageModeShared = CPU + GPU accèdent au même backing
        // Alignement 256 bytes requis par Metal pour les compute shaders
        let alignedBytes = (byteCount + 255) & ~255
        guard let buf = device.makeBuffer(
            length: alignedBytes,
            options: [.storageModeShared]
        ) else {
            throw SharedBufferError.allocationFailed
        }

        self.mtlBuffer = buf
        self.shape = shape
        self.dtype = dtype

        // Calcul des strides row-major
        var s: [Int] = [1]
        for i in (1..<shape.count).reversed() {
            s.insert(s[0] * shape[i], at: 0)
        }
        self.strides = s.map { NSNumber(value: $0 * elementSize) }
    }

    /// Vue CoreML — AUCUNE copie, pointe directement vers le MTLBuffer
    func coreMLView() throws -> MLMultiArray {
        let ptr = mtlBuffer.contents()
        return try MLMultiArray(
            dataPointer: ptr,
            shape: shape.map { NSNumber(value: $0) },
            dataType: dtype,
            strides: strides,
            deallocator: nil  // MTLBuffer gère la durée de vie
        )
    }

    /// Fence Metal — garantit que le GPU a terminé avant lecture CPU/ANE
    func waitForGPU(_ cmdBuffer: MTLCommandBuffer) {
        cmdBuffer.waitUntilCompleted()
        // Alternative non-bloquante avec completion handler :
        // cmdBuffer.addCompletedHandler { _ in self.aneCanStart.signal() }
    }
}

Contrainte de format — NCHW vs NHWC

SWIFT format conversion (si nécessaire) 
// Metal opère naturellement en NCHW [batch, channels, height, width]
// CoreML/ANE attend NHWC [batch, height, width, channels]
// Pour les transformeurs : [B, seq, heads, d_head] — pas de transposition si
// on organise les dims correctement dès le départ.

// SOLUTION : définir le layout Metal aussi en NHWC pour attention
// Ce n'est pas le défaut MLX — il faut forcer avec .transposed()

extension SharedBuffer {
    /// Transposition légère via Metal kernel — si inévitable
    /// [B, H, S, D] → [B, S, H, D] via permute kernel
    func permuteAsync(
        _ cmdBuf: MTLCommandBuffer,
        perm: [Int],
        pipelineState: MTLComputePipelineState
    ) {
        // Utiliser un kernel Metal léger (~0.05ms)
        // Coût minimal comparé à une copie CPU
        let enc = cmdBuf.makeComputeCommandEncoder()!
        enc.setComputePipelineState(pipelineState)
        enc.setBuffer(mtlBuffer, offset: 0, index: 0)
        // ... dispatch threads
        enc.endEncoding()
    }
}
Pour les transformeurs avec seq ≤ 512, organiser le layout d'origine en [B, seq, heads, d_head] (NHWC-like) évite toute transposition. C'est le choix architectural à faire lors de l'export CoreML.

Dispatcher — Routing Logic

Le dispatcher est le cerveau du système. Il décide à chaque couche, et potentiellement à chaque inférence, quelle unité exécute quoi. La décision est basée sur trois signaux : type d'opération (statique), profil thermique (dynamique), et budget latence (contextuel).

SWIFT Sources/Core/Dispatcher.swift 
import Foundation

// ─── Types ───────────────────────────────────────────────────

enum ComputeUnit { case gpu, ane, gpuAndAne }

struct LayerProfile {
    let layerIdx: Int
    let opType: OpType
    var measuredGPU_ns: UInt64 = 0
    var measuredANE_ns: UInt64 = 0
    var transferCost_ns: UInt64 = 0
}

enum OpType {
    case embedding
    case qkvProjection
    case scaledDotProductAttention
    case ffn
    case layerNorm
    case lmHead

    /// Unité statiquement préférée — avant mesures
    var defaultUnit: ComputeUnit {
        switch self {
        case .embedding:                  return .gpu
        case .qkvProjection:              return .ane  // ANE excelle MatMul modéré
        case .scaledDotProductAttention:  return .gpu  // bandwidth-bound → GPU
        case .ffn:                        return .gpu  // large MatMul → GPU
        case .layerNorm:                  return .ane  // faible arith intensity
        case .lmHead:                     return .gpu  // vocab × d_model grand
        }
    }
}

// ─── Dispatcher ──────────────────────────────────────────────

final class Dispatcher {
    private var profiles: [LayerProfile] = []
    private let thermal: ThermalMonitor
    private var calibrated = false

    // Seuil : si ANE_time + transfer_time > GPU_time * margin → GPU
    private let marginFactor: Double = 1.05

    init(thermal: ThermalMonitor) {
        self.thermal = thermal
    }

    /// Décision de routage pour une couche donnée
    func route(layer: Int, op: OpType) -> ComputeUnit {
        // Phase 1 : warm-up — utiliser defaults statiques
        guard calibrated else { return op.defaultUnit }

        guard layer < profiles.count else { return op.defaultUnit }
        let p = profiles[layer]

        // Coût effectif ANE = temps ANE + transfert IOSurface
        let aneCost  = Double(p.measuredANE_ns + p.transferCost_ns)
        let gpuCost  = Double(p.measuredGPU_ns)

        // Ajustement thermique : si GPU throttle, avantage ANE
        let thermalMult = thermal.gpuThrottleFactor  // 1.0 = normal, >1 = throttled
        let adjustedGPU = gpuCost * thermalMult

        if aneCost < adjustedGPU * marginFactor {
            return .ane
        }
        return .gpu
    }

    /// Calibration — 10 runs warm-up + 50 runs mesurés
    func calibrate(
        mlx: MLXEngine,
        ane: ANEEngine,
        sampleInput: SampleBatch
    ) async {
        // Pour chaque couche et chaque op, mesurer GPU et ANE
        // Stocker dans profiles[]
        // ... (voir Benchmark section)
        calibrated = true
    }
}

LIMITATION RÉELLE

CoreML ne permet pas de passer des MLMultiArray de couche en couche sans re-wrapper à chaque fois. Le overhead est ~0.05ms/couche. Sur 12 couches = 0.6ms fixe. À mesurer contre le gain ANE.

ANE LIMIT — 16K OPS

Un mlpackage CoreML ne peut pas dépasser ~16 000 ops internes (limite ANE hardware M-series). Pour 12 couches d'attention, découper en 2-3 mlpackages séparés.

Pipeline Overlap — La vraie source de gain

L'overlap n'est possible qu'entre couches distinctes, pas à l'intérieur d'une couche. L'attention du bloc N peut tourner sur ANE pendant que le GPU calcule la projection QKV du bloc N+1 (qui ne dépend que du LayerNorm précédent, déjà terminé).

SWIFT Pipeline overlap — async/await pattern 
/// Exécution d'un bloc transformer avec overlap GPU ↔ ANE
/// Précondition : QKV du bloc courant déjà calculé sur GPU
func transformerBlockOverlap(
    hidden: MLXArray,
    weights: BlockWeights,
    nextWeights: BlockWeights?,
    layerIdx: Int
) async throws -> MLXArray {

    // ① LayerNorm sur GPU (fast, ~0.1ms)
    let normed = gpu.layerNorm(hidden, w: weights.norm1_w, b: weights.norm1_b)

    // ② QKV projection — GPU pour la couche courante
    let (q, k, v) = gpu.qkvProject(normed, weights: weights.attn)

    // ③ Transférer Q,K,V vers SharedBuffer pour ANE
    let qBuf = try SharedBuffer(device: device, shape: q.shape)
    q.copyToBuffer(qBuf.mtlBuffer)  // GPU memcpy — très rapide
    // Idem K, V

    // ④ LANCER ANE ASYNC — ne pas attendre
    async let attnResult = ane.attentionAsync(
        query: try qBuf.coreMLView(),
        key:   try kBuf.coreMLView(),
        value: try vBuf.coreMLView()
    )

    // ⑤ PENDANT CE TEMPS — GPU calcule QKV du bloc SUIVANT
    var nextQKV: (q: MLXArray, k: MLXArray, v: MLXArray)?
    if let nw = nextWeights {
        let nextNormed = gpu.layerNorm(hidden, w: nw.norm1_w, b: nw.norm1_b)
        nextQKV = gpu.qkvProject(nextNormed, weights: nw.attn)
        // GPU libre pour la prochaine itération
    }

    // ⑥ ATTENDRE l'ANE — normalement déjà fini
    let attn = try await attnResult

    // ⑦ Projection de sortie + résidu — GPU
    let attnMLX = mlxFromCoreML(attn)  // wrap sans copie
    let proj    = gpu.outProject(attnMLX, w: weights.attn.out)
    let res1    = hidden + proj

    // ⑧ FFN — GPU
    let normed2 = gpu.layerNorm(res1, w: weights.norm2_w, b: weights.norm2_b)
    let ffnOut  = gpu.ffnForward(hidden: normed2,
                                   w1: weights.ffn.w1,
                                   w2: weights.ffn.w2,
                                   w3: weights.ffn.w3)
    return res1 + ffnOut
}
Le gain réel de l'overlap dépend du ratio t_ANE / t_GPU_next_QKV. Si t_ANE ≫ t_GPU_QKV, le GPU attend quand même. Mesurer ce ratio en calibration et activer l'overlap seulement si le gain estimé dépasse le coût de synchronisation (~0.1ms de fence overhead).

Thermal Monitor — Awareness sans kext

Accéder aux capteurs SMC directement (températures GPU, ANE) nécessite des entitlements privés Apple ou un kext sur macOS. En userspace sans entitlement, c'est impossible proprement. Les solutions "IOKit SMC" en open-source utilisent des interfaces non documentées qui peuvent casser à chaque mise à jour macOS.

La stratégie réaliste : utiliser deux niveaux de signal, du plus au moins précis.

SWIFT Sources/Core/ThermalMonitor.swift 
import Foundation

final class ThermalMonitor {

    // ── Niveau 1 : ProcessInfo.thermalState (public API) ──────────
    // Granularité : 4 états (nominal, fair, serious, critical)
    // Latence : ~1s pour refléter le changement réel
    // Fiable, documenté, disponible sur iOS aussi

    private var thermalObserver: NSObjectProtocol?

    var gpuThrottleFactor: Double {
        switch ProcessInfo.processInfo.thermalState {
        case .nominal:  return 1.0   // GPU à pleine vitesse
        case .fair:     return 1.15  // Légère dégradation
        case .serious:  return 1.45  // GPU throttle ~30%
        case .critical: return 2.0   // Basculer massivement vers ANE
        @unknown default: return 1.0
        }
    }

    // ── Niveau 2 : Mesure de dérive temporelle (proxy indirect) ───
    // Si les 5 dernières mesures GPU montrent une dégradation croissante,
    // inférer un throttling même si ProcessInfo n'a pas encore changé d'état

    private var gpuHistory: [Double] = []
    private let historySize = 5

    func recordGPULatency(_ ms: Double) {
        gpuHistory.append(ms)
        if gpuHistory.count > historySize {
            gpuHistory.removeFirst()
        }
    }

    /// True si la tendance des 5 dernières mesures est à la hausse > 10%
    var isGPUDegrading: Bool {
        guard gpuHistory.count == historySize else { return false }
        let baseline = gpuHistory.prefix(2).reduce(0, +) / 2.0
        let recent   = gpuHistory.suffix(2).reduce(0, +) / 2.0
        return recent > baseline * 1.10
    }

    // ── Notifications ──────────────────────────────────────────────
    func startMonitoring(onChange: @escaping () -> Void) {
        thermalObserver = NotificationCenter.default.addObserver(
            forName: ProcessInfo.thermalStateDidChangeNotification,
            object: nil,
            queue: .main
        ) { _ in onChange() }
    }

    deinit {
        if let obs = thermalObserver {
            NotificationCenter.default.removeObserver(obs)
        }
    }
}
Alternative avancée (macOS seulement, non-appstore) : utiliser powermetrics via un Process() avec les flags --sampler gpu_power,ane_power. Requiert sudo mais donne les vraies puissances. Viable pour un outil de benchmark — pas pour production.
STRATÉGIE RECOMMANDÉE
thermalState == .nominal
Routing calibré, overlap actif. Pleine performance hybride.
thermalState == .fair
Réduire le parallélisme. Laisser ANE prendre plus de couches.
thermalState ≥ .serious
Basculer vers mode ANE-first. Recalibrer le dispatcher toutes les 30s.
NE PAS FAIRE

• Utiliser IOServiceGetMatchingService pour lire SMC — fragile, non documenté, peut générer KERN_PROTECTION_FAILURE sur macOS 14.4+

• Récalibrer le dispatcher plus souvent que toutes les 5 secondes — les transitions thermiques sont lentes et la recalibration coûte ~50ms

• Ignorer isGPUDegrading — le signal ProcessInfo a ~1s de latence, la dérive de latence détecte le throttle plus tôt

06

Benchmark Harness — Mesurer correctement

Les benchmarks mal conduits sont la source principale de désinformation sur ANE vs GPU. Voici les règles strictes pour des mesures valides.

ERREURS CLASSIQUES À ÉVITER

① Mesurer sans warm-up (JIT MLX, compilation CoreML)
② Inclure le chargement des poids dans la mesure
③ Ne tester qu'une seule longueur de séquence
④ Comparer sur M3 Ultra branché seulement
⑤ Utiliser Date() au lieu de mach_absolute_time()
⑥ Ne pas vérifier que CoreML utilise vraiment l'ANE
⑦ Comparer batch=1 GPU vs batch=8 ANE

MÉTHODE CORRECTE

① 20 runs warm-up, puis 100+ runs mesurés
② Mesurer uniquement forward pass, poids déjà en mémoire
③ Tester seq ∈ {32, 64, 128, 256, 512}
④ Tester sur au moins 2 appareils différents
⑤ Utiliser clock_gettime(CLOCK_MONOTONIC)
⑥ Vérifier ANE via Instruments → Core ML Performance
⑦ Même batch size partout, reporter median + p95

SWIFT Sources/Benchmark/BenchmarkRunner.swift 
import Foundation

struct BenchmarkResult {
    let label: String
    let median_ms: Double
    let p95_ms: Double
    let p99_ms: Double
    let stddev_ms: Double
    let seqLen: Int
    let device: String  // "M3 Ultra" etc.
}

final class BenchmarkRunner {

    /// Clock haute résolution — éviter Date() qui a ~1ms de résolution
    private func now_ns() -> UInt64 {
        var ts = timespec()
        clock_gettime(CLOCK_MONOTONIC_RAW, &ts)
        return UInt64(ts.tv_sec) * 1_000_000_000 + UInt64(ts.tv_nsec)
    }

    /// Run benchmark avec warm-up correct
    func run(
        label: String,
        seqLen: Int,
        warmup: Int = 20,
        iters: Int = 100,
        block: () throws -> Void
    ) throws -> BenchmarkResult {

        // Phase warm-up — permet : JIT compilation MLX,
        // pré-chargement poids CoreML sur ANE, cache Metal shaders
        for _ in 0..<warmup { try block() }

        // GC + flush avant mesure
        MLX.synchronize()

        var timings: [Double] = []
        timings.reserveCapacity(iters)

        for _ in 0..<iters {
            let t0 = now_ns()
            try block()
            MLX.synchronize()  // CRUCIAL : attendre fin GPU avant timer stop
            let dt = Double(now_ns() - t0) / 1_000_000.0
            timings.append(dt)
        }

        timings.sort()
        let median = timings[iters / 2]
        let p95    = timings[Int(Double(iters) * 0.95)]
        let p99    = timings[Int(Double(iters) * 0.99)]
        let mean   = timings.reduce(0, +) / Double(iters)
        let variance = timings.map { ($0 - mean) * ($0 - mean) }.reduce(0, +) / Double(iters)
        let stddev = variance.squareRoot()

        return BenchmarkResult(
            label: label,
            median_ms: median,
            p95_ms: p95,
            p99_ms: p99,
            stddev_ms: stddev,
            seqLen: seqLen,
            device: ProcessInfo.processInfo.machineHardwareName
        )
    }

    /// Export CSV pour analyse externe
    func exportCSV(_ results: [BenchmarkResult], to url: URL) throws {
        var csv = "label,seq_len,median_ms,p95_ms,p99_ms,stddev_ms,device\n"
        for r in results {
            csv += "\(r.label),\(r.seqLen),\(r.median_ms),\(r.p95_ms),\(r.p99_ms),\(r.stddev_ms),\(r.device)\n"
        }
        try csv.write(to: url, atomically: true, encoding: .utf8)
    }
}

    var machineHardwareName: String {
        var size: Int = 0
        sysctlbyname("hw.model", nil, &size, nil, 0)
        var machine = [CChar](repeating: 0, count: size)
        sysctlbyname("hw.model", &machine, &size, nil, 0)
        return String(cString: machine)
    }
